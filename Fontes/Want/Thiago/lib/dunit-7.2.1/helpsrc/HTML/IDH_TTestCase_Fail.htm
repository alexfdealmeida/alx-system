Implementation

procedure TTestCase.Fail(msg: string; errorAddr :Pointer = nil);
begin
  if errorAddr = nil then
    raise __FailureExceptionClass.Create(msg) at CallerAddr
  else
    raise __FailureExceptionClass.Create(msg) at errorAddr;
End;
