Implementation

 
function  MakeTestSuites(name :string; const classes :array of TTestCaseClass): ITestSuite;
var
  i :Integer;
begin
   result := TTestSuite.Create(name);
   for i := Low(classes) to High(classes) do begin
      result.addTest(testSuiteOf(classes[i]));
   end;
End;
